  6. jQuery.event.drag version 2.3.0
  7. jQuery.event.drop version 2.3.0
  8. underscore.js version 1.9.1
  9. download.js version 4.2
  10. turf.js
  11. es6string.js
  12. popper.js version 1.14.6
